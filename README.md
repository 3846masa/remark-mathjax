## Install

```
npm i remark @paperist/remark-mathjax
```

## Usage

```js
const remark = require('remark');
const mathjax = require('remark-mathjax');

const ast = remark().use(mathjax).parse(markdown);
console.dir(ast, { depth: null });
```
